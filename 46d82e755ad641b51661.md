def conv(c):
    if c == '0':
        return 0
    return 1

def solve(s1):
    s2 = s1.split('/')
    s3 = []
    for i in s2:
        s = bin(int(i,16))
        s3.append(s[2:])
    if len(s3[0]) > len(s3[1]):
        while len(s3[1]) < len(s3[0]):
            s3[1] = '0' + s3[1]

        while len(s3[0]) < len(s3[1]):
            s3[0] = '0' + s3[0]
    ans = ""

        a = conv(s3[0][y])
